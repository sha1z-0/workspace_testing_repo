| pg_get_functiondef                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                      |
| ----------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------- |
| CREATE OR REPLACE FUNCTION public.calculate_milestone_progress(p_task_id uuid)
 RETURNS integer
 LANGUAGE plpgsql
AS $function$
DECLARE
  total_milestones INTEGER;
  completed_milestones INTEGER;
  progress INTEGER;
BEGIN
  SELECT COUNT(*) INTO total_milestones
  FROM task_milestones
  WHERE task_id = p_task_id;
  
  IF total_milestones = 0 THEN
    RETURN 0;
  END IF;
  
  SELECT COUNT(*) INTO completed_milestones
  FROM task_milestones
  WHERE task_id = p_task_id AND status = 'completed';
  
  progress := ROUND((completed_milestones::DECIMAL / total_milestones::DECIMAL) * 100);
  
  RETURN progress;
END;
$function$

  UPDATE tasks

| update_task_milestone_count  | 
BEGIN
  UPDATE tasks
  SET milestone_count = (
    SELECT COUNT(*) 
    FROM task_milestones 
    WHERE task_id = COALESCE(NEW.task_id, OLD.task_id)
  )
  WHERE id = COALESCE(NEW.task_id, OLD.task_id);
  
  RETURN COALESCE(NEW, OLD);
END;

| calculate_milestone_progress | 
DECLARE
  total_milestones INTEGER;
  completed_milestones INTEGER;
  progress INTEGER;
BEGIN
  SELECT COUNT(*) INTO total_milestones
  FROM task_milestones
  WHERE task_id = p_task_id;
  
  IF total_milestones = 0 THEN
    RETURN 0;
  END IF;
  
  SELECT COUNT(*) INTO completed_milestones
  FROM task_milestones
  WHERE task_id = p_task_id AND status = 'completed';
  
  progress := ROUND((completed_milestones::DECIMAL / total_milestones::DECIMAL) * 100);
  
  RETURN progress;
END;
